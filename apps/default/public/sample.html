WonderCall Demo

This page demonstrates loading and mounting WonderCall. All you need:

<script src="https://assets.wondercall.ai/embeds/default/latest/main.js"></script>
<script>
document.addEventListener('DOMContentLoaded', function() {
    window.WonderCall.mount(
    {
        apiPublicKey: "your-wondercall-api-key",
        assistantId: "your-assistant-id"
    });
});
</script>
